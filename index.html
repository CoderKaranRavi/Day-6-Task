class Person:

def __init__(self, name, age, address):
self.name = name
self.age = age
self.address = address

def display_info(self):
print(f"Name: {self.name}")
print(f"Age: {self.age}")
print(f"Address: {self.address}")

Write a class to calculate the uber price.

class UberPricingCalculator:

BASE_FARE = 120
COST_PER_MILE = 15
COST_PER_MINUTE = 8
SURGE_MULTIPLIER = 1.4
def __init__(self, distance, duration, surge=False):
self.distance = distance
self.duration = duration
self.surge = surge

def calculate_price(self):

Apply surge pricing if applicable

if self.surge:
price_before_surge = self._calculate_base_price()
total_price = price_before_surge * self.SURGE_MULTIPLIER
else:
total_price = self._calculate_base_price()
return round(total_price, 2)

def _calculate_base_price(self):
# Calculate base fare base_fare = self.BASE_FARE

Calculate distance cost

distance_cost = self.distance * self.COST_PER_MILE

Calculate duration cost

duration_cost = self.duration * self.COST_PER_MINUTE

Calculate total price

total_price = base_fare + distance_cost + duration_cost
return total_price
